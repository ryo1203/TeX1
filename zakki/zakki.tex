  m \frac{d^4}{dt^4}x_A & = & k \left( \frac{d^2}{dt^2}x_B - \frac{d^2}{dt^2}x_A \right) \\

  m \D{4}x_A & = & k ( -2k(L - x_A + x_B) ) \\

  m \D{4}x_A & = & -2k \left( m \D{2}x_A - mg \right) \\
  \D{4}x_A & = & -2k \D{2}x_A + 2kg \\
